// --- 介面定義 ---
interface IStorage {
  getLineUser(lineUserId: string): Promise<LineUser | undefined>;
  getAllLineUsers(): Promise<LineUser[]>;
  upsertLineUser(data: InsertLineUser): Promise<LineUser>;
  updateOrder(id: number, data: Partial<Order>): Promise<Order>;
}
// --- 實作 ---
// 查詢單一 LINE 使用者
async getLineUser(lineUserId: string) {
  const [user] = await db
    .select()
    .from(lineUsers)
    .where(eq(lineUsers.lineUserId, lineUserId));
  return user;
}
// 取得所有 LINE 使用者列表（給後台搜尋用）
async getAllLineUsers() {
  return db.select().from(lineUsers);
}
// 新增或更新 LINE 使用者（使用者每次 LINE 登入時呼叫）
async upsertLineUser(data: InsertLineUser) {
  const existing = await this.getLineUser(data.lineUserId);
  if (existing) {
    // 已存在 → 更新資料
    const [updated] = await db
      .update(lineUsers)
      .set({
        displayName: data.displayName,
        pictureUrl: data.pictureUrl,
        venueKey: data.venueKey
      })
      .where(eq(lineUsers.lineUserId, data.lineUserId))
      .returning();
    return updated;
  }
  // 不存在 → 新增
  const [created] = await db
    .insert(lineUsers)
    .values(data)
    .returning();
  return created;
}

完整運作流程
前提：客人用新 LINE 帳號在系統上登入過一次
       → 系統自動存入 line_users 表
管理員操作：
  1. 進入後台 → 訂單查詢
  2. 找到需要重新綁定的訂單 → 點編輯
  3. 在「LINE 帳號綁定」區塊：
     - 看到目前綁定的 LINE 帳號（頭像+名稱）
     - 在搜尋框輸入客人的新 LINE 名稱
     - 從列表中點選正確的帳號
  4. 點「儲存變更」
結果：
  - 訂單的 lineUserId 更新為新帳號
  - 之後的 LINE 推播通知會發到新帳號
  - 客人用新帳號登入「我的訂單」可以看到這筆訂單
